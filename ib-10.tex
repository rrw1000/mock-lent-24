\begin{question}[MockIB,year=2024,paper=1,question=10,author=rrw]{Computation Theory}

\triposset{fullmarks=20}
\topic{Computation Theory}

\begin{enumerate}
\item Describe a register machine \fullmarks{3}
\item Show how it is possible to encode a register machine program as a number suitable for storage in a register machine register \fullmarks{5}
\item Give an informal description of how you might write a register machine program, $P$, which given an encoding $d$ of a register machine program in register 0, computes the result of running the register machine program encoded by $d$. \fullmarks{4}
\item Hence, or otherwise, determine whether it is possible to write a register machine program $Q$ that decides if two other register machine programs $A$ and $B$ produce the same output in register 0, and produce a proof of your determination. \fullmarks{8}
\end{enumerate}

\end{question}
